53  public static Stream GetEmbeddedResourceStream(Assembly assembly, string resourceFileName)
54  {
55  var resourceNames = assembly.GetManifestResourceNames();
56 
57  var resourcePaths = resourceNames
58  .Where(x => x.EndsWith(resourceFileName, StringComparison.CurrentCultureIgnoreCase))
59  .ToArray();
60 
61  if (!resourcePaths.Any())
62  {
63  throw new Exception(string.Format("Resource ending with {0} not found.", resourceFileName));
64  }
65 
66  if (resourcePaths.Count() > 1)
67  {
68  throw new Exception(string.Format("Multiple resources ending with {0} found: {1}{2}", resourceFileName, Environment.NewLine, string.Join(Environment.NewLine, resourcePaths)));
69  }
70 
71  return assembly.GetManifestResourceStream(resourcePaths.Single());
72  }
