172  internal byte[] GetHashFinalBlock(byte[] input, int ibStart, int cbSize, ABCDStruct ABCD, Int64 len)
173  {
174  byte[] working = new byte[64];
175  byte[] length = BitConverter.GetBytes(len);
176 
177  //Padding is a single bit 1, followed by the number of 0s required to make size congruent to 448 modulo 512. Step 1 of RFC 1321
178  //The CLR ensures that our buffer is 0-assigned, we don't need to explicitly set it. This is why it ends up being quicker to just
179  //use a temporary array rather then doing in-place assignment (5% for small inputs)
180  Array.Copy(input, ibStart, working, 0, cbSize);
181  working[cbSize] = 0x80;
182 
183  //We have enough room to store the length in this chunk
184  if (cbSize < 56)
185  {
186  Array.Copy(length, 0, working, 56, 8);
187  GetHashBlock(working, ref ABCD, 0);
188  }
189  else //We need an aditional chunk to store the length
190  {
191  GetHashBlock(working, ref ABCD, 0);
192  //Create an entirely new chunk due to the 0-assigned trick mentioned above, to avoid an extra function call clearing the array
193  working = new byte[64];
194  Array.Copy(length, 0, working, 56, 8);
195  GetHashBlock(working, ref ABCD, 0);
196  }
197  byte[] output = new byte[16];
198  Array.Copy(BitConverter.GetBytes(ABCD.A), 0, output, 0, 4);
199  Array.Copy(BitConverter.GetBytes(ABCD.B), 0, output, 4, 4);
200  Array.Copy(BitConverter.GetBytes(ABCD.C), 0, output, 8, 4);
201  Array.Copy(BitConverter.GetBytes(ABCD.D), 0, output, 12, 4);
202  return output;
